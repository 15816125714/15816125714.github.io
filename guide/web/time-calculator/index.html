时间计算器 ​

说明 ​

实例采用 Momentjs Api 实现，通过可视化界面方便、快捷的计算出结果，支持链式调用、本地缓存记录等操作。

效果展示 ​

vue
<template>
<ClientOnly>
  <div class="operation-wrap">
    <div>操作</div>
    <div class="operation-options">
      <div class="item">
        <div class="label">折叠</div>
        <el-switch
          v-model="collapseForm"
          size="small"
          style="--el-switch-on-color: var(--vp-c-brand);"
          @change="handleCollapseForm" />
      </div>
      <div class="item reset-btn" @click="handleReset">
        <el-icon :class="['icon', isReset && 'reset-loading']">
          <RefreshLeft />
        </el-icon>
        &nbsp;重置
      </div>
    </div>
  </div>
  <ItemGroup
    v-for="(item, index) in configs.listData"
    :key="item.key"
    :index="index"
    :configs="item"
    :len="configs.listData.length"
    @add-time="(child, childIndex) => handleAddTime(item, index, child, childIndex)"
    @remove-time="(child, childIndex) => handleRemoveTime(item, index, child, childIndex)"
    @func-change="(val, childIndex) => handleFuncChange(item, index, val, childIndex)"
    @operation="(type) => handleOperation(index, type)"
    @submit="(res) => handleSubmit(item, index, res)"
    @update-storage="updateStorage" />
</ClientOnly>
</template>

<script setup>
import { ref, reactive } from "vue";
import { ElMessage } from "element-plus";
import { RefreshLeft } from "@element-plus/icons-vue";
import moment from "moment";
import { momentFuncParams, argsOptions } from "./config";
import ItemGroup from "./ItemGroup.vue";
import pkg from "lodash";
const { cloneDeep } = pkg;

const collapseForm = ref(false);
const opts = reactive(Object.keys(moment.prototype || {}));

const fnConfig = {
  label: "函数",
  component: "select",
  isFunc: true,
  identifier: "func",
  value: "",
  placeholder: "请选择",
  options: opts,
};

const defaultConf = {
  title: "",
  timeRange: false,
  showResult: false,
  keywords: "",
  datePickerType: "datetime",
  collapseForm: false,
  key: +new Date(),
  options: [
    {
      label: "时间",
      value: "datetime",
      identifier: "datePickerType",
      component: "radio",
      options: [
        {
          label: "默认",
          value: "datetime",
        },
        {
          label: "年",
          value: "year",
        },
        {
          label: "月",
          value: "month",
        },
        {
          label: "日",
          value: "date",
        },
        {
          label: "多日",
          value: "dates",
        },
      ],
    },
    {
      label: "",
      component: "date-picker",
      identifier: "startTime",
      value: "",
      placeholder: "默认当前时间",
    },
    {
      ...cloneDeep(fnConfig),
    },
  ]
};

let configs = reactive({
  listData: [],
});
const init = () => {
  configs.listData = [cloneConfigs()];
};
init();

function cloneConfigs (cloneObj, assignData = {}) {
  const cloneData = cloneDeep(cloneObj || defaultConf);
  cloneData.key = +new Date();
  return Object.assign({}, cloneData, assignData);
};

function handleCollapseForm (val) {
  configs.listData.forEach((row) => {
    row.collapseForm = val;
  });
};

const isReset = ref(false);
const handleReset = () => {
  isReset.value = true;
  setTimeout(() => {
    isReset.value = false;
  }, 1000);
  localStorage.removeItem("time-calculator");
  init();
  ElMessage({
    message: "重置成功！",
    type: "success",
    duration: 1000,
  });
};

const localConfigs = localStorage.getItem("time-calculator");
if (localConfigs) {
  configs = reactive(JSON.parse(localConfigs));
}

const getFuncParams = (func) => {
  const matches = /[^(]+([^)]*)?/gm.exec(Function.prototype.toString.call(func));
  if (matches[1]) {
    var res = matches[1].replace(/[^,\w]*/g, "").split(",").filter((row) => row);
    return res.length ? res : null;
  }
  return null;
}

const handleAddTime = (item, index, child, childIndex) => {
  configs.listData[index].options.splice(childIndex + 1, 0, {
    label: "",
    component: "date-picker",
    identifier: "endTime",
    value: "",
    placeholder: "结束时间",
    isEndTime: true,
  });
  configs.listData[index].timeRange = true;
  updateStorage();
};

const handleRemoveTime = (item, index, child, childIndex) => {
  configs.listData[index].options.splice(childIndex, 1);
  configs.listData[index].timeRange = false;
  updateStorage();
};

const handleFuncChange = (item, parentIndex, method, index) => {
  const funcParams = momentFuncParams[method] || [];
  configs.listData[parentIndex].options = item.options.slice(0, index + 1);
  if (!funcParams.length) return;
  const callChaining = moment()[method]() instanceof moment;
  configs.listData[parentIndex].options.push({
    label: "参数",
    value: "",
    children: funcParams.map((text) => {
      const options = argsOptions(text);
      return {
        label: "",
        placeholder: text,
        value: "",
        component: options ? "select" : "input",
        options: options ? options : [],
      };
    }),
  });
  if (callChaining) {
    configs.listData[parentIndex].options.push(cloneDeep(fnConfig));
  }
  updateStorage();
};

function updateStorage () {
  localStorage.setItem("time-calculator", JSON.stringify(configs));
};

const handleSubmit = (item, index, res) => {
  updateStorage();
};

const handleOperation = (index, type) => {
  if (["add"].includes(type)) {
    configs.listData.splice(index + 1, 0, cloneConfigs());
  } else if (["clone"].includes(type)) {
    const cloneData = cloneConfigs(configs.listData[index], {});
    cloneData.title += " - 克隆";
    configs.listData.splice(index + 1, 0, cloneData);
  } else if (["moveup"].includes(type)) {
    const obj = cloneDeep(configs.listData[index]);
    configs.listData.splice(index, 1);
    configs.listData.splice(index - 1, 0, obj);
  } else if (["movedown"].includes(type)) {
    const obj = cloneDeep(configs.listData[index]);
    configs.listData.splice(index, 1);
    configs.listData.splice(index + 1, 0, obj);
  } else {
    // 刪除
    configs.listData.splice(index, 1);
  }
  updateStorage();
};
</script>
