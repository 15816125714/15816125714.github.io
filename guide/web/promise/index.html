vue
<template>
  <div class="item">{{ (time / 1000) }}秒后，随机数大于{{ comparisonValue }}表示成功，否则失败</div>
  <template v-if="randomNum1">
    <div class="item">　值：{{ randomNum1 }}</div>
    <div class="item">结果：
      <el-button link :type="randomNum1 > comparisonValue ? 'success' : 'danger'">{{ resultMsg }}</el-button>
    </div>
  </template>
  <div class="item">
    <el-button v-if="!randomNum1" type="primary" :loading="!randomNum1">{{ !randomNum1 ? "请稍后" : resultMsg }}</el-button>
    <el-button v-else type="primary" @click="handleInit">
      <el-icon :class="['icon']">
        <RefreshLeft />
      </el-icon>&nbsp;重试
    </el-button>
  </div>
</template>

<script setup>
import { ref } from "vue";
import { RefreshLeft } from "@element-plus/icons-vue";
import { myPromise } from "./core.js";

const time = ref(1000);
const comparisonValue = ref(0.5);
const randomNum1 = ref(0);
const resultMsg = ref("");

const handleInit = () => {
  randomNum1.value = "";
  resultMsg.value = "";
  const prom = new myPromise((resolve, reject) => {
    setTimeout(() => {
      randomNum1.value = Math.random();
      if (randomNum1.value > comparisonValue.value) {
        resolve(randomNum1.value);
        resultMsg.value = "成功";
      } else {
        reject(randomNum1.value);
        resultMsg.value = "失败";
      }
    }, time.value);
  })
  .then((res) => {
    console.log("myPromise then ------------ ", res);
    return new myPromise((resolve, reject) => {
      setTimeout(() => {
        resolve();
      }, 1000);
    });
  }
  , (e) => {
    console.log("myPromise catch ------------ ", e);
  }
  ).then((res) => {
    console.log("myPromise then2 ----------- ");
  });
};

handleInit();

const PENDING = 'PENDING';
const FULFILLED = 'FULFILLED';
const REJECTED = 'REJECTED';

const resolvePromise = (prom, x, resolve, reject) => {
  if (prom === x) {
    return reject(new TypeError('Chaining cycle detected for promise #<Promise>'));
  }
  let called;
  if ((typeof x === "object" && x != null) || typeof x === "function") {
    try {
      let then = x.then;
      if (typeof then === "function") {
        then.call(x, y => {
          if (called) return;
          called = true;
          resolvePromise(prom, y, resolve, reject);
        }, r => {
          if (called) return;
          called = true;
          reject(r);
        });
      } else {
        resolve(x);
      }
    } catch (e) {
      if (called) return;
      called = true;
      reject(e);
    }
  } else {
    resolve(x);
  }
};

export class myPromise {
  constructor(func) {
    this.status = PENDING;
    this.value = null;
    this.onFulfilledCallbacks = [];
    this.onRejectedCallbacks = [];
    const resolve = (val) => {
      if (this.status === PENDING) {
        this.status = FULFILLED;
        this.value = val;
        this.onFulfilledCallbacks.forEach(fn => fn());
      }
    }
    const reject = (val) => {
      if (this.status === PENDING) {
        this.status = REJECTED;
        this.value = val;
        this.onRejectedCallbacks.forEach(fn => fn());
      }
    }
    try {
      func(resolve, reject);
    } catch (e) {
      reject(e);
    }
  }
  then(onFulfilled, onRejected) {
    onFulfilled = typeof onFulfilled === "function" ? onFulfilled : fn => fn;
    onRejected = typeof onRejected === "function" ? onRejected : err => { throw err };

    let prom = new myPromise((resolve, reject) => {
      // 基础版
      // switch(this.status) {
      //   case PENDING:
      //     this.onFulfilledCallbacks.push(() => {
      //       onFulfilled(this.value);
      //     });
      //     this.onRejectedCallbacks.push(() => {
      //       onRejected(this.value);
      //     });
      //     break;
      //   case FULFILLED:
      //     onFulfilled(this.value);
      //     break;
      //   case REJECTED:
      //     onRejected(this.value);
      //     break;
      // }

      // 升级版
      switch(this.status) {
        case PENDING:
          this.onFulfilledCallbacks.push(() => {
            setTimeout(() => {
              try {
                let x = onFulfilled(this.value);
                resolvePromise(prom, x, resolve, reject);
              } catch (e) {
                reject(e);
              };
            })
          });
          this.onRejectedCallbacks.push(() => {
            setTimeout(() => {
              try {
                let x = onRejected(this.value);
                resolvePromise(prom, x, resolve, reject);
              } catch (e) {
                reject(e);
              };
            })
          });
          break;
        case FULFILLED:
          setTimeout(() => {
            try {
              let x = onFulfilled(this.value);
              resolvePromise(prom, x, resolve, reject);
            } catch (e) {
              reject(e);
            }
          }, 0);
          break;
        case REJECTED:
          setTimeout(() => {
            try {
              let x = onRejected(this.value);
              resolvePromise(prom, x, resolve, reject);
            } catch (e) {
              reject(e);
            }
          }, 0);
          break;
      }
    });
    return prom;
  }
};
