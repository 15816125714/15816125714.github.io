<template>
  <div
    ref="wrap"
    :class="['sc-list-more', moreIndex && 'show-more']">
    <template v-for="(item, index) in filterListData">
      <div
        v-if="(!moreIndex || index < moreIndex)"
        ref="listItems"
        :key="index"
        :style="itemStyle"
        class="list-item">
        <div
          :title="item.label"
          class="text">{{ item.label }}</div>
      </div>
    </template>

    <!-- 更多 -->
    <el-popover
      placement="bottom-start"
      trigger="click">
      <template #reference>
        <div
          v-show="showMore"
          ref="moreBtn"
          class="list-item more-btn">
          <span class="text">...</span>
        </div>
      </template>
      <div class="more-content">
        <div
          v-for="(item, index) in moreList"
          :key="index"
          :title="item.label"
          :style="itemStyle"
          class="list-item">
          <div style="color: rgba(173, 201, 217, 1)" class="text">{{ item.label }}</div>
        </div>
      </div>
    </el-popover>
  </div>
</template>

<script setup>
import { ref, onMounted, onBeforeUnmount, computed, watch, nextTick } from "vue";

const props = defineProps({
  listData: {
    type: Array,
    default: () => ([])
  },
  itemStyle: {
    type: Object,
    default: () => ({})
  },
  onlyShowOne: {
    type: Boolean,
    default: false
  }
})
const listItems = ref(null);
const showMore = ref(false);
const moreIndex = ref(0);
const moreList = ref([]);
onMounted(() => {
  init()
  window.addEventListener('resize', () => init());
});
onBeforeUnmount(() => {
  window.removeEventListener('resize', () => init());
});

watch(() => props.listData, () => {
  init();
});

const filterListData = computed(() => {
  return props.listData.map((row) => {
    if (typeof row === 'string') {
      return { label: row };
    } else {
      return row;
    }
  });
});

function init() {
  handleReset();
  if (props.onlyShowOne) {
    if (filterListData.value.length > 1) {
      handleSet(1);
    }
    return;
  }
  nextTick(() => {
    listItems.value && listItems.value.forEach((el, index) => {
      if (el.offsetTop && !moreIndex.value) {
        handleSet(index);
      }
    });
  });
}

function handleReset() {
  showMore.value = false;
  moreIndex.value = 0;
  moreList.value = [];
}
function handleSet(index) {
  showMore.value = true;
  moreIndex.value = index;
  moreList.value = filterListData.value.slice(index);
}
</script>
