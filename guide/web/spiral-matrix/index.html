      <div v-for="(child, idx) in item" :key="`child-${idx}`" :class="['child', activeNum === child.value && 'active']">{{ icons[child.value] }}</div>

<script setup>
import { onMounted, ref } from 'vue';
import { spiralMatrix } from "./core";
const isAnimate = ref(false);
const baseNum = 7;
const list = spiralMatrix(baseNum);
const options = ref(list);
const icons = ref(['🐶','🐺','🐱','🐭','🐹','🐰','🐸','🐯','🐨','🐻','🐷','🐽','🐮','🐗','🐵','🐒','🐴','🐑','🐘','🐼','🐧','🐦','🐤','🐥','🐣','🐔','🐍','🐢','🐛','🐝','🐜','🐞','🐌','🐙','🐚','🐠','🐟','🐬','🐳','🐋','🐄','🐏','🐀','🐃','🐅','🐇','🐉','🐎','🐐','🐓','🐕','🐖','🐁','🐂','🐲','🐡','🐊','🐫','🐪','🐆','🐈','🐩','🐾','💐','🌸','🌷','🍀','🌹','🌻','🌺','🍁','🍃','🍂','🌿','🌾','🍄','🌵','🌴','🌲','🌳','🌰','🌱','🌼','🌐','🌞','🌝','🌚','🌑','🌒','🌓','🌔','🌕','🌖','🌗','🌘','🌜','🌛','🌙','🌍','🌎','🌏','🌋','🌌','🌠','⭐','☀','⛅','☁','⚡','☔','❄','⛄','🌀','🌁','🌈','🌊']);
const activeNum = ref(0);
let timer = null;

function count() {
  activeNum.value++;
  if (activeNum.value >= baseNum * baseNum) {
    activeNum.value = 1;
  }
}

function startAnimate() {
  isAnimate.value = true;
  timer = setInterval(() => count(), 50);
  setTimeout(() => {
    clearInterval(timer);
    setTimeout(() => count(), 200);
    setTimeout(() => count(), 400);
    setTimeout(() => count(), 700);
    setTimeout(() => {
      count();
      isAnimate.value = false;
    }, 1000);
  }, 2000 + Math.ceil(Math.random() * 3000));
}
</script>
