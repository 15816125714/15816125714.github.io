<script>
import { spiralMatrix } from "./core";
export default {
  data() {
    return {
      num: 3,
      options: [],
      visibleIndex: 1,
    }
  },
  created() {
    this.init();
  },
  methods: {
    init() {
      this.options = spiralMatrix(this.num);
      let timer = setInterval(() => {
        this.visibleIndex++;
        if (this.visibleIndex > this.num * this.num) {
          clearInterval(timer);
        }
      }, 15);
      console.log("螺旋矩阵 ---------------- ", spiralMatrix(this.num));
    },
  },
};
</script>

export const spiralMatrix = (n) => {
  if (isNaN(n) || n < 1) return;
  let len = n * n;
  let res = Array.from({length: n}, () => Array(n));
  let target = 0;
  let index = -1;
  let minRow = 0;
  let maxRow = n - 1;
  let minColumn = 0;
  let maxColumn = n - 1;
  let fx = "right";
  let isCorner = false;
  for(let i = 0; i < len; i++) {
    isCorner = false;
    if (fx === "right") {
      index++;
      if (index >= maxRow) {
        fx = "down";
        minColumn++;
        isCorner = true;
      }
    } else if (fx === "down") {
      target++;
      if (target >= maxColumn) {
        fx = "left";
        maxRow--;
        isCorner = true;
      }
    } else if (fx === "left") {
      index--;
      if (index <= minRow) {
        fx = "up";
        maxColumn--;
        isCorner = true;
      }
    } else if (fx === "up") {
      target--;
      if (target <= minColumn) {
        fx = "right";
        minRow++
        isCorner = true;
      }
    }
    res[target][index] = { fx, value: i+1, isEnd: i === len - 1, isCorner };
  }
  return res
}
